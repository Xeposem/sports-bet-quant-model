3. Create `src/model/bayesian.py`:

   ```python
   """Bayesian hierarchical logistic model with PyMC — surface partial pooling."""
   from __future__ import annotations
   import logging
   import numpy as np
   from src.model.base import (
       build_training_matrix, compute_time_weights, temporal_split, LOGISTIC_FEATURES,
   )

   logger = logging.getLogger(__name__)

   SURFACE_INDEX = {"Hard": 0, "Clay": 1, "Grass": 2}

   def _surface_str_to_idx(surface_strings):
       """Convert surface string array to integer index array. Unknown -> 0 (Hard)."""
       return np.array([SURFACE_INDEX.get(s, 0) for s in surface_strings], dtype=np.int64)

   def train_fold(X_train, y_train, X_val, y_val, w_train,
                  surface_train=None, config=None):
       """Train Bayesian hierarchical logistic on pre-split data.

       Parameters
       ----------
       X_train, y_train, X_val, y_val: arrays
       w_train: sample weights (not used by PyMC — here for interface compat)
       surface_train: array of surface index integers (0=Hard, 1=Clay, 2=Grass).
           If None, defaults to all 0 (Hard) — happens in walk-forward where
           surface info is not in the differential feature matrix.
       config: dict with optional keys: draws (default 2000), tune (default 1000),
           chains (default 2), target_accept (default 0.9)

       Returns
       -------
       dict with keys: "model" (PyMC model), "idata" (InferenceData),
       plus "metrics" dict with max_rhat, converged (bool)
       """
       import pymc as pm
       import arviz as az

       if config is None:
           config = {}
       draws = config.get("draws", 2000)
       tune = config.get("tune", 1000)
       chains = config.get("chains", 2)
       target_accept = config.get("target_accept", 0.9)

       if surface_train is None:
           surface_train = np.zeros(len(y_train), dtype=np.int64)

       # Standardize features for better MCMC convergence
       X_mean = X_train.mean(axis=0)
       X_std = X_train.std(axis=0)
       X_std[X_std == 0] = 1.0  # avoid division by zero
       X_scaled = (X_train - X_mean) / X_std

       with pm.Model() as model:
           X_data = pm.Data("X", X_scaled)
           surf_data = pm.Data("surface_idx", surface_train)

           # Hyperpriors for partial pooling across surfaces
           mu_alpha = pm.Normal("mu_alpha", 0, 1)
           sigma_alpha = pm.HalfNormal("sigma_alpha", 1)

           # Surface-specific intercepts (partial pooling)
           alpha = pm.Normal("alpha", mu=mu_alpha, sigma=sigma_alpha, shape=3)

           # Global feature weights
           beta = pm.Normal("beta", 0, 1, shape=X_train.shape[1])

           # Logistic link
           logit_p = alpha[surf_data] + pm.math.dot(X_data, beta)
           p = pm.Deterministic("p", pm.math.sigmoid(logit_p))

           # Likelihood
           obs = pm.Bernoulli("obs", p=p, observed=y_train)

           idata = pm.sample(
               draws=draws, tune=tune, chains=chains,
               target_accept=target_accept, progressbar=False,
               random_seed=42,
           )

       # Check convergence
       summary = az.summary(idata, var_names=["alpha", "beta"])
       max_rhat = float(summary["r_hat"].max())
       converged = max_rhat < 1.1

       if not converged:
           logger.warning("Bayesian fold convergence warning: max r_hat=%.3f", max_rhat)

       # Compute validation Brier score using p50 predictions
       val_preds = _predict_internal(model, idata, X_val, X_mean, X_std,
                                      np.zeros(len(y_val), dtype=np.int64))
       from sklearn.metrics import brier_score_loss, log_loss
       val_brier = float(brier_score_loss(y_val, val_preds["p50"]))
       val_logloss = float(log_loss(y_val, np.clip(val_preds["p50"], 1e-15, 1-1e-15)))

       metrics = {
           "max_rhat": max_rhat,
           "converged": converged,
           "val_brier_score": val_brier,
           "val_log_loss": val_logloss,
       }

       return {
           "model": model,
           "idata": idata,
           "X_mean": X_mean,
           "X_std": X_std,
           "metrics": metrics,
       }

   def _predict_internal(model, idata, X_new, X_mean, X_std, surface_idx_new):
       """Internal prediction using pm.set_data + sample_posterior_predictive."""
       import pymc as pm
       import arviz as az

       X_scaled = (X_new - X_mean) / X_std

       with model:
           pm.set_data({"X": X_scaled, "surface_idx": surface_idx_new})
           ppc = pm.sample_posterior_predictive(
               idata, var_names=["p"], progressbar=False,
           )

       # Extract posterior predictive samples for p
       p_samples = ppc.posterior_predictive["p"].values  # shape: (chains, draws, n_matches)
       p_flat = p_samples.reshape(-1, p_samples.shape[-1])  # (chains*draws, n_matches)

       p5 = np.percentile(p_flat, 5, axis=0)
       p50 = np.percentile(p_flat, 50, axis=0)
       p95 = np.percentile(p_flat, 95, axis=0)

       return {"p5": p5, "p50": p50, "p95": p95}

   def predict(trained, features, surface_idx=None):
       """Registry-compliant predict.

       Parameters
       ----------
       trained: dict from train_fold with model, idata, X_mean, X_std
       features: np.ndarray of shape (n_features,) or (1, n_features)
       surface_idx: int or array of surface indices. Default 0 (Hard).

       Returns
       -------
       dict with p5, p50, p95 (floats for single match), calibrated_prob (=p50)
       """
       if features.ndim == 1:
           features = features.reshape(1, -1)
       if surface_idx is None:
           surface_idx = np.zeros(features.shape[0], dtype=np.int64)
       elif isinstance(surface_idx, int):
           surface_idx = np.array([surface_idx], dtype=np.int64)

       result = _predict_internal(
           trained["model"], trained["idata"],
           features, trained["X_mean"], trained["X_std"],
           surface_idx,
       )

       # For single-match prediction, return scalars
       if features.shape[0] == 1:
           return {
               "p5": float(result["p5"][0]),
               "p50": float(result["p50"][0]),
               "p95": float(result["p95"][0]),
               "calibrated_prob": float(result["p50"][0]),
           }
       return {
           "p5": result["p5"],
           "p50": result["p50"],
           "p95": result["p95"],
           "calibrated_prob": result["p50"],
       }

   def train(conn, config=None):
       """Registry-compliant train: builds matrix from DB, trains Bayesian model."""
       if config is None:
           config = {}
       X, y, match_dates = build_training_matrix(conn)
       weights = compute_time_weights(match_dates)
       split = temporal_split(X, y, weights, match_dates)
       trained = train_fold(
           split["X_train"], split["y_train"],
           split["X_val"], split["y_val"],
           split["w_train"],
           config=config,
       )
       return trained, trained.pop("metrics")

   def save_bayesian_model(trained, path):
       """Save Bayesian model idata to NetCDF (not joblib)."""
       trained["idata"].to_netcdf(path)
   ```

   - SURFACE_INDEX constant at module level for consistent train/predict encoding
   - `target_accept=0.9` for better acceptance rate on small samples
